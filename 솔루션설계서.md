### 2.3 StrategyEvaluator — condition_tree 구조

condition_tree는 JSON으로 DB에 저장되며 재귀적으로 평가된다.

```json
{
  "operator": "AND",
  "conditions": [
    {
      "indicator": "RSI",
      "params": {"period": 14, "timeframe": "1h"},
      "operator": "lt",
      "value": 30
    },
    {
      "operator": "OR",
      "conditions": [
        {
          "indicator": "VOLUME",
          "params": {"timeframe": "1h"},
          "operator": "gt_multiple",
          "value": 2.0,
          "compare_to": "volume_ma_20"
        },
        {
          "indicator": "BB",
          "params": {"period": 20, "std": 2, "timeframe": "1h"},
          "operator": "price_below_lower"
        }
      ]
    }
  ]
}
```

```python
# backend/app/trading/strategy_evaluator.py
from dataclasses import dataclass
from typing import Any
import pandas as pd
import ta

@dataclass
class ConditionResult:
    matched: bool
    triggered: list[str]

class StrategyEvaluator:
    SUPPORTED_INDICATORS = {"RSI", "MACD", "BB", "MA", "EMA", "STOCH", "CCI", "VOLUME"}

    def evaluate(self, condition_tree: dict, ohlcv: pd.DataFrame) -> ConditionResult:
        indicators = self._compute_indicators(condition_tree, ohlcv)
        return self._eval_tree(condition_tree, indicators)

    def _eval_tree(self, node: dict, indicators: dict) -> ConditionResult:
        if "operator" in node and node["operator"] in ("AND", "OR"):
            # 논리 연산자 노드
            results = [self._eval_tree(c, indicators) for c in node["conditions"]]
            if node["operator"] == "AND":
                matched = all(r.matched for r in results)
            else:
                matched = any(r.matched for r in results)
            triggered = [t for r in results for t in r.triggered if r.matched]
            return ConditionResult(matched=matched, triggered=triggered)
        else:
            # 리프 조건 노드
            return self._eval_leaf(node, indicators)

    def _eval_leaf(self, cond: dict, indicators: dict) -> ConditionResult:
        key = f"{cond['indicator']}_{cond['params'].get('timeframe','1h')}"
        value = indicators.get(key)
        if value is None:
            return ConditionResult(False, [])

        op = cond["operator"]
        threshold = cond.get("value")
        matched = {
            "lt":  lambda v, t: v < t,
            "gt":  lambda v, t: v > t,
            "lte": lambda v, t: v <= t,
            "gte": lambda v, t: v >= t,
            "gt_multiple": lambda v, t: v > indicators.get(cond["compare_to"], 0) * t,
            "price_below_lower": lambda v, _: indicators["CLOSE"] < v["lower"],
            "price_above_upper": lambda v, _: indicators["CLOSE"] > v["upper"],
            "golden_cross": lambda v, _: v["cross"] == "golden",
            "dead_cross":   lambda v, _: v["cross"] == "dead",
        }.get(op, lambda v, t: False)(value, threshold)

        label = f"{cond['indicator']} {op} {threshold}"
        return ConditionResult(matched=matched, triggered=[label] if matched else [])

    def _compute_indicators(self, tree: dict, ohlcv: pd.DataFrame) -> dict:
        """조건 트리에서 필요한 지표만 계산"""
        needed = self._extract_needed_indicators(tree)
        result = {"CLOSE": float(ohlcv["close"].iloc[-1])}

        for ind, params in needed.items():
            tf = params.get("timeframe", "1h")
            close = ohlcv["close"]
            volume = ohlcv["volume"]

            if ind == "RSI":
                period = params.get("period", 14)
                result[f"RSI_{tf}"] = float(ta.momentum.rsi(close, period).iloc[-1])
            elif ind == "MACD":
                macd = ta.trend.MACD(close)
                prev_diff = macd.macd_diff().iloc[-2]
                curr_diff = macd.macd_diff().iloc[-1]
                cross = "golden" if prev_diff < 0 <= curr_diff else \
                        "dead" if prev_diff > 0 >= curr_diff else "none"
                result[f"MACD_{tf}"] = {"diff": float(curr_diff), "cross": cross}
            elif ind == "BB":
                period = params.get("period", 20)
                std = params.get("std", 2)
                bb = ta.volatility.BollingerBands(close, period, std)
                result[f"BB_{tf}"] = {
                    "upper": float(bb.bollinger_hband().iloc[-1]),
                    "lower": float(bb.bollinger_lband().iloc[-1]),
                    "mid":   float(bb.bollinger_mavg().iloc[-1]),
                }
            elif ind == "VOLUME":
                result[f"VOLUME_{tf}"] = float(volume.iloc[-1])
                result["volume_ma_20"] = float(volume.rolling(20).mean().iloc[-1])
        return result

    def _extract_needed_indicators(self, node: dict, acc=None) -> dict:
        if acc is None:
            acc = {}
        if "conditions" in node:
            for c in node["conditions"]:
                self._extract_needed_indicators(c, acc)
        elif "indicator" in node:
            key = node["indicator"]
            if key not in acc:
                acc[key] = node.get("params", {})
        return acc
```

```python
# backend/app/trading/ai_consultant.py
import json, asyncio
from anthropic import AsyncAnthropic

VALID_DECISIONS   = {"execute", "hold", "avoid"}
VALID_RISK_LEVELS = {"low", "medium", "high"}
CACHE_TTL = 300  # 5분

class AIConsultant:
    def __init__(self, redis_client, settings):
        self.client = AsyncAnthropic(api_key=settings.ANTHROPIC_API_KEY)
        self.redis = redis_client
        self.timeout = settings.AI_CONSULT_TIMEOUT_SECONDS

    async def get_cached_advice(self, strategy_id: str) -> dict | None:
        """Redis 캐시에서 최신 자문 결과 조회"""
        raw = await self.redis.get(f"ai:advice:{strategy_id}")
        return json.loads(raw) if raw else None

    async def refresh_advice(self, strategy: dict, market_ctx: dict) -> dict:
        """Claude API 호출 → 검증 → Redis 캐시 저장"""
        prompt = self._build_prompt(strategy, market_ctx)
        try:
            async with asyncio.timeout(self.timeout):
                response = await self.client.messages.create(
                    model="claude-opus-4-6",
                    max_tokens=1024,
                    thinking={"type": "adaptive"},
                    output_config={
                        "format": {
                            "type": "json_schema",
                            "schema": {
                                "type": "object",
                                "properties": {
                                    "decision":     {"type": "string", "enum": list(VALID_DECISIONS)},
                                    "confidence":   {"type": "integer", "minimum": 0, "maximum": 100},
                                    "reason":       {"type": "string"},
                                    "risk_level":   {"type": "string", "enum": list(VALID_RISK_LEVELS)},
                                    "key_concerns": {"type": "array", "items": {"type": "string"}},
                                },
                                "required": ["decision", "confidence", "reason", "risk_level"],
                                "additionalProperties": False,
                            }
                        }
                    },
                    messages=[{"role": "user", "content": prompt}],
                )
        except asyncio.TimeoutError:
            # 타임아웃 시 이전 캐시 유지 (혹은 None 반환)
            return await self.get_cached_advice(strategy["id"])

        text = next(b.text for b in response.content if b.type == "text")
        result = json.loads(text)

        # 화이트리스트 검증 (구조화 출력이지만 추가 방어)
        if result["decision"] not in VALID_DECISIONS:
            raise ValueError(f"AI 응답 오염: decision={result['decision']}")
        if result["risk_level"] not in VALID_RISK_LEVELS:
            raise ValueError(f"AI 응답 오염: risk_level={result['risk_level']}")

        await self.redis.setex(
            f"ai:advice:{strategy['id']}", CACHE_TTL, json.dumps(result)
        )
        return result

    def _build_prompt(self, strategy: dict, ctx: dict) -> str:
        # 블라인드 모드: 신호 방향 미포함 (확증 편향 방지)
        return f"""
당신은 암호화폐 리스크 분석 전문가입니다.
아래 시장 데이터를 분석하고 JSON으로 응답하세요.

## 분석 대상
심볼: {ctx['symbol']}
현재가: {ctx['price']} {ctx['quote_currency']}
24시간 변동률: {ctx['change_24h']}%

## 기술 지표 (현재)
{json.dumps(ctx['indicators'], ensure_ascii=False, indent=2)}

## 최근 체결 내역 (5건)
{json.dumps(ctx['recent_trades'], ensure_ascii=False, indent=2)}

## 시장 컨텍스트
- BTC 도미넌스: {ctx.get('btc_dominance', 'N/A')}%
- 전체 시장 방향: {ctx.get('market_trend', 'N/A')}

현재 시장 상황에서 거래 실행 여부에 대한 의견을 제시하세요.
신호 방향은 제공하지 않습니다. 순수하게 시장 상황만 평가하세요.
"""
```

### 2.5 긴급 정지 분산 락 시퀀스

```python
# backend/app/tasks/trading_tasks.py
from redis.asyncio import Redis
import asyncio

async def emergency_stop(strategy_id: str, user_id: str, reason: str, db, redis: Redis, exchange):
    lock_key = f"emergency:lock:{strategy_id}"
    flag_key = f"emergency:stop:{strategy_id}"

    # Step 1: 분산 락 획득 (원자적 SET NX EX)
    acquired = await redis.set(lock_key, user_id, nx=True, ex=5)
    if not acquired:
        return {"status": "already_stopping"}

    try:
        # Step 2: DB 전략 비활성화
        await db.execute(
            "UPDATE strategies SET is_active=false WHERE id=:sid",
            {"sid": strategy_id}
        )
        await db.commit()

        # Step 3: Redis 긴급 정지 플래그 (1시간 TTL)
        await redis.setex(flag_key, 3600, reason)

        # Step 4: 미체결 주문 일괄 취소 (병렬)
        open_orders = await db.fetch_all(
            "SELECT * FROM orders WHERE strategy_id=:sid AND status='open'",
            {"sid": strategy_id}
        )
        cancel_results = await asyncio.gather(
            *[exchange.cancel_order(o["exchange_order_id"], o["symbol"]) for o in open_orders],
            return_exceptions=True
        )
        await db.execute(
            "UPDATE orders SET status='cancelled' WHERE strategy_id=:sid AND status='open'",
            {"sid": strategy_id}
        )

        # Step 5: WebSocket 상태 Push
        await redis.publish(
            f"strategy:status:{strategy_id}",
            json.dumps({"type": "emergency_stop", "strategy_id": strategy_id, "reason": reason})
        )

        # Step 6: DB 이력 저장
        await db.execute(
            "INSERT INTO emergency_stops (strategy_id, user_id, reason, cancelled_orders) VALUES (:sid, :uid, :r, :co)",
            {"sid": strategy_id, "uid": user_id, "r": reason, "co": json.dumps([o["id"] for o in open_orders])}
        )
        await db.commit()

        return {"status": "stopped", "cancelled_orders": len(open_orders)}

    finally:
        # Step 7: 분산 락 해제 (항상 실행)
        await redis.delete(lock_key)
```

```python
    - 체결 가격: bar[i] 시가 (실제 체결 가능 시점)
    """
    def __init__(self, strategy: dict, config: dict):
        self.strategy = strategy
        self.evaluator = StrategyEvaluator(strategy)
        self.initial_capital = Decimal(str(config["initial_capital"]))
        self.commission = Decimal(str(config.get("commission_pct", 0.001)))
        self.slippage  = Decimal(str(config.get("slippage_pct", 0.0005)))

    async def run(self, ohlcv_df: pd.DataFrame) -> dict:
        capital = self.initial_capital
        equity  = []
        trades  = []
        position = None

        WARMUP = 200   # 지표 안정화 구간

        for i in range(WARMUP, len(ohlcv_df)):
            # 지표 계산: i 이전 데이터만 사용 (룩어헤드 방지)
            window = ohlcv_df.iloc[:i]
            indicators = self.evaluator._compute_indicators(self.strategy["condition_tree"], window)

            current_open = Decimal(str(ohlcv_df["open"].iloc[i]))
            current_close = Decimal(str(ohlcv_df["close"].iloc[i]))

            if position is None:
                result = self.evaluator._eval_tree(self.strategy["condition_tree"], indicators)
                if result.matched:
                    entry_price = current_open * (1 + self.slippage)
                    amount = capital * Decimal("0.95") / entry_price
                    fee = amount * entry_price * self.commission
                    capital -= fee
                    position = {
                        "entry_price": entry_price,
                        "entry_time": ohlcv_df.index[i],
                        "amount": amount,
                        "high_water_mark": current_close,
                    }
            else:
                # TP/SL/Trailing 체크
                pnl_pct = (current_close - position["entry_price"]) / position["entry_price"]
                sl = Decimal(str(self.strategy.get("stop_loss_pct", -0.02)))
                tp = Decimal(str(self.strategy.get("take_profit_pct", 0.04)))

                exit_result = self.evaluator._eval_tree(
                    self.strategy.get("exit_condition", {}), indicators
                )
                should_exit = exit_result.matched or pnl_pct <= sl or pnl_pct >= tp

                if should_exit:
                    exit_price = current_open * (1 - self.slippage)
                    fee = position["amount"] * exit_price * self.commission
                    realized = position["amount"] * (exit_price - position["entry_price"]) - fee
                    capital += position["amount"] * exit_price - fee
                    trades.append({
                        "entry_time": str(position["entry_time"]),
                        "exit_time": str(ohlcv_df.index[i]),
                        "pnl": float(realized),
                        "pnl_pct": float(pnl_pct),
                        "won": realized > 0,
                    })
                    position = None

            equity.append(float(capital))

        return self._compute_metrics(equity, trades)

    def _compute_metrics(self, equity: list, trades: list) -> dict:
        eq = np.array(equity, dtype=float)
        returns = np.diff(eq) / np.maximum(eq[:-1], 1e-10)
        peak = np.maximum.accumulate(eq)
        mdd  = float(((eq - peak) / np.maximum(peak, 1e-10)).min()) * 100
        wins = [t for t in trades if t["won"]]
        losses = [t for t in trades if not t["won"]]
        gross_profit = sum(t["pnl"] for t in wins) or 1e-10
        gross_loss   = abs(sum(t["pnl"] for t in losses)) or 1e-10

        return {
            "final_capital": float(eq[-1]),
            "total_return_pct": float((eq[-1] - eq[0]) / eq[0] * 100) if eq[0] else 0,
            "max_drawdown_pct": mdd,
            "sharpe_ratio": float(returns.mean() / (returns.std() + 1e-10) * np.sqrt(252)),
            "win_rate": len(wins) / max(len(trades), 1),
            "profit_factor": gross_profit / gross_loss,
            "total_trades": len(trades),
            "winning_trades": len(wins),
            "losing_trades": len(losses),
            "equity_curve": [{"i": i, "v": v} for i, v in enumerate(equity[::10])],
            "trade_history": trades,
        }
```
